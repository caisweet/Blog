管道流简例

2016/6/22 posted in  Java

2016年06月22日

问题

用管道流读写文件

思路

  1. 创建管道输入输出流，并关联
  2. 实现Runable接口创建两个线程任务分别接入输入输出流
  3. 创建线程跑任务
s=>start: 开始
e=>end: 结束
o=>operation: 创建管道输入输出流，并关联
op=>operation: 实现Runable接口创建两个线程任务分别接入输入输出流
ope=>operation: 创建线程跑任务
s->o->op->ope->e

代码

import java.io.IOException;
import java.io.PipedInputStream;
import java.io.PipedOutputStream;

public class Java20160622_3 {
    public static void main(String []args) throws IOException {
        PipedInputStream input=new PipedInputStream();
        PipedOutputStream output=new PipedOutputStream();
        input.connect(output);
        new Thread(new Input(input)).start();
        new Thread(new Output(output)).start();
    }
    //继承Runable接口来实现多线程
    static class Input implements Runnable{
        private PipedInputStream in;
        Input(PipedInputStream in){
            this.in=in;
        }
        public void run(){
                    //创建缓冲区来接收数据
            byte[] buf=new byte[1024];
            try {
            //判断数据量
                int len=in.read(buf);
                //创建字符串接收数据
                String s=new String(buf,0,len);
                System.out.println("s="+s);
            } catch (IOException e) {
                e.printStackTrace();
            }finally {
                try {
                    in.close();
                } catch (IOException e) {
                    e.printStackTrace();
                }
            }
        }
    }
    static class Output implements Runnable{
        private PipedOutputStream out;
        Output(PipedOutputStream out){
            this.out=out;
        }
        public void run(){
            try {
            //输出数据
                out.write("我爱你".getBytes());
            } catch (IOException e) {
                e.printStackTrace();
            }finally {
                try {
                    out.close();
                } catch (IOException e) {
                    e.printStackTrace();
                }
            }
        }
    }
}
