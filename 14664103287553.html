获取指定后缀名文件的绝对路径

2016/6/20 posted in  Java

2016年06月20日

问题

想要知道某个文件夹下面是否有需要的后缀名文件，有就把文件的绝对路径记录下来，然后打印在文本文件中。

思路

  1. 遍历目录
  2. 判断文件后缀名，是就把文件添加到数组中
  3. 把数组中的文件打印出绝对路径到文本文件中
st=>start: 开始
end=>end: 结束
op=>operation: 遍历目录
ope=>operation: 记录文件
co=>condition: 是否目录？
cond=>condition: 是否要找的后缀名文件？
st->op->co(yes)->op
co(no)->cond(yes)->ope->end

代码


import java.io.*;
import java.util.ArrayList;
import java.util.List;

public class Java20160620_2 {
    public static void main(String []args) throws IOException {
         File dir=new File("/test");
         //用匿名内部类的方式重写过滤器，过滤以.mp3结尾的文件
        FilenameFilter filter=new FilenameFilter() {
            @Override
            public boolean accept(File dir, String name) {
                return name.endsWith(".mp3");
            }
        };
        //创建一个集合来存储过滤后的文件
        List<File> list=new ArrayList<File>();
        getFiles(dir,filter,list);
        wirteIt(list);
    }

    public static void wirteIt(List<File> list) throws IOException {
        File f=new File("music.txt");
        //创建缓冲区来写文件
        BufferedWriter fw=new BufferedWriter(new FileWriter(f));
        for (File file:list){
        //把集合中存储的对象，获取绝对路径后写入
           fw.append(file.getAbsolutePath());
            fw.newLine();
            fw.flush();
        }
        fw.close();
    }

    public static void getFiles(File dir, FilenameFilter filter, List<File> list) {
        File[] files=dir.listFiles();
        //遍历目录，是目录就递归调用自身继续向下级目录遍历
        for (File file:files){
            if (file.isDirectory()){
                getFiles(file,filter,list);
            }else{
            //遍历到文件就获取名字用过滤器判断是否符合，符合就添加到集合中
               if(filter.accept(file,file.getName())){
                   list.add(file);
               }
            }
        }
    }
}
